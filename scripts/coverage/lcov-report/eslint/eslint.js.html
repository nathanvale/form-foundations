const util = require('util');
const exec = util.promisify(require('child_process').exec);
const path = require('path');
async function runESLint() {
  const options = { maxBuffer: 500 * 1024 };
  const eslintConfigPath = path.join(__dirname, '../.eslintrc.json');
  const eslintIgnorePath = path.join(__dirname, '../.eslintignore');
 
  return exec(
    `eslint -c ${eslintConfigPath} "packages/style-guide/dist/**/*.js" "packages/originate/dist/**/*.js"  -f json`,
    options
  )
    .then(response => response.stdout)
    .catch(e => {
      if (e.stderr === '') return e.stdout;
      throw e;
    });
}
 
exports.runESLint = runESLint;
