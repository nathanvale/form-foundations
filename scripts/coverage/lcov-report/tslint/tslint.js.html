function intersect(files, patterns) {
  let intersection = [];
  patterns.forEach(pattern => {
    intersection = [
      ...intersection,
      ...minimatch.match(files, pattern, { matchBase: true, dot: true })
    ];
  });
  return [...new Set(intersection)];
}
